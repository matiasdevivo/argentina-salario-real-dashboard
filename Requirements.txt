# ipc.xlsx
#   Contenido : Variación mensual del IPC Nacional
#   Fuente    : INDEC (Instituto Nacional de Estadística y Censos)
#   URL       : https://www.indec.gob.ar
#   Columnas  : Fecha (Date), Variacion_Mensual_IPC (Decimal)
#   Rango     : Enero 2017 — Agosto 2025

# Remuneracion_-_Filtrada.xlsx
#   Contenido : RIPTE mensual (Remuneración Imponible Promedio de Trabajadores Estables)
#   Fuente    : MTEySS (Ministerio de Trabajo)
#   URL       : https://www.argentina.gob.ar/trabajo/estadisticas/ripte
#   Columnas  : Fecha (Date), Salario_Nominal (Entero)
#   Rango     : Enero 2017 — Agosto 2025

# cambio_tipo.xlsx
#   Contenido : Tipo de cambio nominal promedio mensual (pesos por dólar)
#   Fuente    : BCRA (Banco Central de la República Argentina)
#   URL       : https://www.bcra.gob.ar
#   Columnas  : Fecha (texto → convertir a Date en Power Query), TC_Nominal (Decimal)
#   Rango     : Enero 2017 — Agosto 2025
#   Nota      : Requiere transformación de fecha en Power Query (ver docs/)
